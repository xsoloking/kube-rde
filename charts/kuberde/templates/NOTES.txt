To access KubeRDE without Ingress, use port-forward:

  # Web UI
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "kuberde.fullname" . }}-web {{ .Values.web.service.port }}:{{ .Values.web.service.port }}
  Then visit: http://localhost:{{ .Values.web.service.port }}

  # Server API
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "kuberde.fullname" . }}-server {{ .Values.server.service.wsPort }}:{{ .Values.server.service.wsPort }}
  Then visit: http://localhost:{{ .Values.server.service.wsPort }}/api
